// Firebase Firestore Güvenlik Kuralları
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Blog yazıları (posts) koleksiyonu için kurallar
    match /posts/{post} {
      // Herkes okuyabilir
      allow read: if true;
      
      // Yazma, güncelleme ve silme işlemleri için kimlik doğrulama gerekir (ileride eklenecek)
      allow write, update, delete: if true; // Test için geçici olarak herkese açık
    }
    
    // Yorumlar (comments) koleksiyonu için kurallar
    match /comments/{comment} {
      // Herkes okuyabilir
      allow read: if true;
      
      // Yeni yorum eklemek için gerekli alanlar olmalı
      allow create: if request.resource.data.keys().hasAll(['name', 'content', 'slug']) 
                      && request.resource.data.name != '' 
                      && request.resource.data.content != '';
      
      // Yorum silme ve güncelleme izni yok
      allow update, delete: if false;
    }
  }
}
