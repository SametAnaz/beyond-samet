// Firebase Firestore Güvenlik Kuralları
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Blog yazıları (posts) koleksiyonu için kurallar
    match /posts/{post} {
      // Herkes okuyabilir
      allow read: if true;
      
      // Yazma, güncelleme ve silme işlemleri için kimlik doğrulama gerekir (ileride eklenecek)
      allow write, update, delete: if true; // Test için geçici olarak herkese açık
    }
    
    // Yorumlar (comments) koleksiyonu için kurallar
    match /comments/{comment} {
      // Herkes okuyabilir
      allow read: if true;
      
      // Yeni yorum eklemek için gerekli alanlar olmalı
      allow create: if 
        // Gerekli alanlar
        request.resource.data.keys().hasAll(['name', 'content', 'slug', 'createdAt', 'consentGiven']) 
        // Boş olmaması gereken alanlar
        && request.resource.data.name != '' 
        && request.resource.data.content != ''
        // Onay verilmiş olmalı
        && request.resource.data.consentGiven == true
        // Maksimum uzunluk
        && request.resource.data.name.size() <= 100
        && request.resource.data.content.size() <= 2000
        // E-posta varsa geçerli bir formatta olmalı
        && (!('email' in request.resource.data) 
            || request.resource.data.email == null 
            || request.resource.data.email.matches('^[^@]+@[^@]+\\.[^@]+$'));
      
      // Yorum silme ve güncelleme izni yok
      allow update, delete: if false;
    }
  }
}
